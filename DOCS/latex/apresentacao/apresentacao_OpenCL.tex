\section{Introdução}
\subsection{Descrição do Problema}
\frame
{
	\frametitle{Descrição do Problema}
	\begin{itemize}
		\item O \textbf{\textcolor{Blue}{transporte coletivo}} têm fundamental importância no cotidiano da população.
		\begin{itemize}
			\pause
			\item Mais econômico.
			\pause
			\item Contribui com a diminuição da poluição do ar e sonora.
			\pause						
			\item Solução para problemas de congestionamento de veículos.
		\end{itemize}
	\end{itemize}
}

\frame
{
	\frametitle{Descrição do Problema}
	\begin{itemize}
		\item A \textbf{\textcolor{Red}{falta de informação}} sobre o transporte público ainda é um grande problema.
		\begin{itemize}
			\pause
			\item Afeta os \textbf{\textcolor{Blue}{moradores}} da cidade.
			\pause
			\item Afeta os \textbf{\textcolor{Blue}{turistas}} que pretendem utilizar o transporte coletivo.
		\end{itemize}
		\vskip10pt
		\pause
		\item Expectativa de 200 mil turistas durante a Copa em Manaus. (Bernardo Monteiro, coordenador da UGP Copa)
	\end{itemize}
}

\frame
{
	\frametitle{Descrição do Problema}
	\begin{itemize}
		\item \textbf{\textcolor{Red}{Não}} há um lugar onde o usuário possa ter informações básicas sobre o percurso de um ônibus.
		\pause
		\item Paradas de ônibus \textbf{\textcolor{Red}{não}} informam as linhas de ônibus que passam por ela.
	\end{itemize}
}

{
	\frametitle{Descrição do Problema}

\frame
{
	\frametitle{Descrição do Problema}
	\begin{itemize}
		\item O usuário não sabe ao certo o horário em que deve chegar às paradas e nem os horários exatos dos ônibus.
		\pause
		\item Muito tempo é \textbf{\textcolor{Red}{perdido}} diariamente esperando um determinado ônibus.
		\begin{itemize}
			\pause
			\item Média de 1 hora e 30 minutos (Pesquisa com 11 usuários de transporte coletivo)
		\end{itemize}
	\end{itemize}
}

\subsection{Justificativa}
\frame
{
	\frametitle{Justificativa}
	\begin{itemize}
		\item Contribuir para a criação de um modelo de transporte mais eficiente e de melhor \textbf{\textcolor{Blue}{qualidade}}, uma vez que o transporte público está passando por um processo de \textbf{\textcolor{Red}{descrédito}} cada vez maior junto ao público alvo.
	\end{itemize}
}

\subsection{Objetivo Geral}
\frame
{
	\frametitle{Objetivo Geral}
	\begin{itemize}
		\item Desenvolver um sistema \textit{back-end} e \textit{front-end} que permita a visualização das informações do transporte coletivo em um aplicativo para plataforma Android.
		\begin{itemize}
				\pause
				\item Percusos das linhas de ônibus da cidade.
				\pause
				\item Visualização em tempo real de uma determinada linha de ônibus.
		\end{itemize}
	\end{itemize}
}

\subsection{Objetivos Específicos}
\frame
{
	\frametitle{Objetivos Específicos}
	\begin{itemize}
		\item Desenvolver um \textbf{\textcolor{Blue}{Sistema Embarcado}}, no qual irá utilizar as tecnologias GPS e GSM/GPRS com o objetivo de enviar informações em tempo real da localização do ônibus para um serviço web.
		\pause
		\item Desenvolver um \textbf{\textcolor{Blue}{Aplicativo Android}} que fornecerá informações detalhadas do percurso de determinada linha de ônibus e sua posição em tempo real para o usuário.
		\pause
		\item Desenvolver um \textbf{\textcolor{Blue}{Serviço Web}} que irá gerenciar as requisições provenientes dos módulos embarcados e módulos clientes.
	\end{itemize}
}
